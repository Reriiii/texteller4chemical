# Server commands for TexTeller + RFL-MSD experiments
# Assumes Linux shell, uv environment is ready, and external tools are cloned:
#   external/GraphMatchingTool/eval.py
#   external/RFL-MSD/RFL/RFL.py or external/RFL-MSD/RFL.py

export CUDA_VISIBLE_DEVICES=1
export HF_HOME=/home/tuancuong/data5t/chemteller/.hf_cache
export HF_HUB_CACHE=/home/tuancuong/data5t/chemteller/.hf_cache/hub
export TOKENIZERS_PARALLELISM=false


# 0) Check CUDA / PyTorch environment
uv run python -c "import torch; print(torch.__version__); print(torch.cuda.is_available()); print(torch.version.cuda); print(torch.cuda.get_device_name(0) if torch.cuda.is_available() else 'NONE')"


# 1) Materialize official EDU-CHEMC splits with graph-normalized SSML target
uv run python scripts/materialize_hf_edu_chemc.py \
  --dataset_id ConstantHao/EDU-CHEMC_MM23 \
  --out_dir data/processed/edu_chemc_graph_norm \
  --target_field ssml_graph_norm


# 2) Validate ssml_graph_norm preserves ssml_normed graph semantics
uv run python scripts/validate_graph_norm.py \
  --dataset_dir data/processed/edu_chemc_graph_norm \
  --splits train validation test \
  --source_key ssml_normed \
  --normalized_key ssml_graph_norm \
  --graph_matching_tool_dir external/GraphMatchingTool \
  --graph_num_workers 8 \
  --out_dir outputs/reports/graph_norm_validation_full


# 3) Materialize full RFL-MSD dataset from graph-norm prepared dataset
# Do not use --on_error fallback for training; it mixes RFL and non-RFL labels.
uv run python scripts/create_rfl_target_dataset.py \
  --dataset_dir data/processed/edu_chemc_graph_norm \
  --out_dir data/processed/edu_chemc_rfl_msd \
  --splits train validation test \
  --source_key ssml_normed \
  --fallback_source_keys ssml_sd chemfig chemfg \
  --target_field ssml_rfl \
  --graph_label_field ssml_rfl_graph_norm \
  --rfl_tool_dir external/RFL-MSD \
  --num_workers 8 \
  --overwrite \
  --on_error skip


# 4) Pre-train diagnostics for full RFL-MSD data
# Pass condition: no blockers and oracle graph EM/structure EM = 1.0 on sampled rows.
uv run python scripts/evaluate_rfl_msd_pretrain.py \
  --dataset_dir data/processed/edu_chemc_rfl_msd \
  --splits train validation test \
  --max_rows_per_split 5000 \
  --oracle_graph_samples 200 \
  --rfl_tool_dir external/RFL-MSD \
  --graph_matching_tool_dir external/GraphMatchingTool \
  --graph_num_workers 8 \
  --pretrained_model_name_or_path OleehyO/TexTeller \
  --out_report outputs/reports/rfl_msd_pretrain_eval_full_sample.json


# 5) Smoke train: full custom TexTeller encoder + MSD decoder on small subset
uv run accelerate launch \
  --num_machines 1 \
  --num_processes 1 \
  --mixed_precision bf16 \
  --dynamo_backend no \
  scripts/train_texteller_msd.py \
  --config configs/train_edu_chemc_texteller_msd.yaml \
  --dataset_dir data/processed/edu_chemc_rfl_msd \
  --pretrained_model_name_or_path OleehyO/TexTeller \
  --output_dir outputs/runs/smoke_texteller_msd_64 \
  --max_train_samples 64 \
  --max_eval_samples 64


# 6) Evaluate smoke custom MSD checkpoint
uv run python scripts/evaluate_texteller_msd.py \
  --model_ckpt outputs/runs/smoke_texteller_msd_64/best \
  --dataset_dir data/processed/edu_chemc_rfl_msd \
  --config configs/train_edu_chemc_texteller_msd.yaml \
  --split validation \
  --batch_size 1 \
  --num_beams 1 \
  --max_new_tokens 256 \
  --dtype bf16 \
  --output_csv outputs/eval_smoke_texteller_msd_64_validation_greedy.csv \
  --max_samples 50 \
  --graph_eval \
  --graph_matching_tool_dir external/GraphMatchingTool \
  --graph_label_key ssml_rfl_graph_norm \
  --graph_num_workers 8 \
  --rfl_tool_dir external/RFL-MSD


# 7) Full experiment: custom TexTeller encoder + MSD decoder
uv run accelerate launch \
  --num_machines 1 \
  --num_processes 1 \
  --mixed_precision bf16 \
  --dynamo_backend no \
  scripts/train_texteller_msd.py \
  --config configs/train_edu_chemc_texteller_msd.yaml \
  --dataset_dir data/processed/edu_chemc_rfl_msd \
  --pretrained_model_name_or_path OleehyO/TexTeller \
  --output_dir outputs/runs/edu_chemc_texteller_msd_full_custom_bf16


# 8) Full evaluation: custom MSD best checkpoint on test split
uv run python scripts/evaluate_texteller_msd.py \
  --model_ckpt outputs/runs/edu_chemc_texteller_msd_full_custom_bf16/best \
  --dataset_dir data/processed/edu_chemc_rfl_msd \
  --config configs/train_edu_chemc_texteller_msd.yaml \
  --split test \
  --batch_size 1 \
  --num_beams 1 \
  --max_new_tokens 1024 \
  --dtype bf16 \
  --output_csv outputs/eval_texteller_msd_full_custom_bf16_test_greedy.csv \
  --graph_eval \
  --graph_matching_tool_dir external/GraphMatchingTool \
  --graph_label_key ssml_rfl_graph_norm \
  --graph_num_workers 8 \
  --rfl_tool_dir external/RFL-MSD


# 9) Optional comparison: stable TexTeller seq2seq on ssml_graph_norm target
uv run accelerate launch \
  --num_processes 1 \
  --num_machines 1 \
  --mixed_precision bf16 \
  --dynamo_backend no \
  scripts/train.py \
  --config configs/train_edu_chemc.yaml \
  --dataset_dir data/processed/edu_chemc_graph_norm \
  --pretrained_model_name_or_path OleehyO/TexTeller \
  --output_dir outputs/runs/edu_chemc_texteller_graph_norm_full_model_bf16_30ep


# 10) Optional comparison eval: stable graph-norm checkpoint
uv run python scripts/evaluate.py \
  --model_ckpt outputs/runs/edu_chemc_texteller_graph_norm_full_model_bf16_30ep/best \
  --dataset_dir data/processed/edu_chemc_graph_norm \
  --split test \
  --batch_size 8 \
  --num_beams 1 \
  --max_new_tokens 1024 \
  --dtype bf16 \
  --output_csv outputs/eval_graph_norm_full_model_bf16_30ep_test_greedy.csv \
  --graph_eval \
  --graph_matching_tool_dir external/GraphMatchingTool \
  --graph_label_key ssml_graph_norm \
  --graph_num_workers 8
